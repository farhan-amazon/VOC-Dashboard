async function fetchData() { const response = await fetch('https://docs.google.com/spreadsheets/d/e/2PACX-1vQK9aXwzV8gK6R9o3qY5W4x8Drk4r9Qn4WZzP7xG6r8qR3Jhz52uG1p3O7H4H0n5B2Hg7c5pXy5W8h3Jq/pub?output=csv'); const data = await response.text(); return parseCSV(data); } function parseCSV(data) { const rows = data.split('\n').map(row => row.split(',')); return rows; } let rawData = []; async function initialize() { rawData = await fetchData(); populateFilters(); updateCharts(); } function populateFilters() { const operatorSelect = document.getElementById('operator'); const storeCodeSelect = document.getElementById('storeCode'); const complaintSelect = document.getElementById('complaint'); const rcaCategorySelect = document.getElementById('rcaCategory'); // Populate filters based on unique values in the dataset // Assuming first row is headers const operators = [...new Set(rawData.slice(1).map(row => row[0]))]; operators.forEach(op => operatorSelect.add(new Option(op, op))); storeCodes.forEach(code => storeCodeSelect.add(new Option(code, code))); complaints.forEach(comp => complaintSelect.add(new Option(comp, comp))); rcaCategories.forEach(cat => rcaCategorySelect.add(new Option(cat, cat))); } function filterData() { const operator = document.getElementById('operator').value; const storeCode = document.getElementById('storeCode').value; const date = document.getElementById('date').value; const week = document.getElementById('week').value; const complaint = document.getElementById('complaint').value; const rcaCategory = document.getElementById('rcaCategory').value; return rawData.filter(row => { return (!operator || row[0] === operator) && (!storeCode || row[1] === storeCode) && (!date || row[2] === date) && (!week || row[3] == week) && (!complaint || row[6] === complaint) && (!rcaCategory || row[8] === rcaCategory); }); } function updateCharts() { const filteredData = filterData(); // Set up data for the charts // Example: Bar Chart (you can customize based on your needs) const labels = filteredData.map(row => row[1]); const pieChartCtx = document.getElementById('pieChart').getContext('2d'); new Chart(barChartCtx, { type: 'bar', { labels: labels, datasets: [{ label: 'Complaints Overview', values, backgroundColor: 'rgba(75, 192, 192, 0.2)', borderColor: 'rgba(75, 192, 192, 1)', borderWidth: 1 }] }, options: { scales: { y: { beginAtZero: true } }, plugins: { tooltip: { callbacks: { label: function(tooltipItem) { return `Complaints: ${tooltipItem.raw}`; } } } } } }); // You can create more charts (lineChart, pieChart) similarly } initialize();
